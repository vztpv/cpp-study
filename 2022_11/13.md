
    std::vector<bool> vs std::vector<int>
  
```C++
{
    std::vector<bool> vec;
    int a = 0, b = 0;

    a = clock();
    vec.reserve(MAX);
    for (size_t i = 0; i < MAX; ++i) {
      vec.push_back(i % 2 == 0);
    }
    b = clock();
    std::cout << vec.size() << " " << b - a << "ms\n";
}
```
